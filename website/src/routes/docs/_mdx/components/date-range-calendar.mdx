---
title: 范围日期选择
description: 直接展示在页面中的日期范围选择组件，支持选择开始和结束日期。
---

## 演示

```jsx demo online=DateRangeCalendar componentProps=calendar/date-range-calendar.tsx|color,size,readOnly,disabled,dualPanel,showWeekNumbers
import {
  DateRangeCalendar,
  CalendarPrev,
  CalendarNext,
  CalendarTitle,
  CalendarGrid,
} from "@resolid/react-ui";

export default function App() {
  return (
    <DateRangeCalendar>
      <div className="flex gap-1">
        <CalendarPrev />
        <CalendarTitle />
        <CalendarNext />
      </div>
      <CalendarGrid />
    </DateRangeCalendar>
  );
}
```

## 用法

```js
import {
  DateRangeCalendar,
  CalendarTitle,
  CalendarPrev,
  CalendarNext,
  CalendarGrid,
} from "@resolid/react-ui";
```

- `DateRangeCalendar`: 包含范围日期选择的所有部分。
- `CalendarPrev`: 日历导航按钮。它根据当前日历视图，导航一个月/一年/十年前的日历。
- `CalendarNext`: 日历导航按钮。它根据当前日历视图，导航一个月/年/十年后的日历。
- `CalendarTitle`: 日历的当前日期标题部分。
- `CalendarGrid`: 日历的网格部分。

```jsx
<DateRangeCalendar>
  <div className="flex">
    <CalendarPrev />
    <CalendarTitle />
    <CalendarNext />
  </div>
  <CalendarGrid />
</DateRangeCalendar>
```

## 特点

- 完整的键盘导航
- 可控值与默认值
- 完整焦点管理
- 本地化支持
- 高度可组合性

## 举例

### 可控值

```jsx demo
import {
  DateRangeCalendar,
  CalendarPrev,
  CalendarNext,
  CalendarTitle,
  CalendarGrid,
} from "@resolid/react-ui";
import { useState } from "react";

export default function App() {
  const [value, setValue] = useState("2026-03-12 to 2026-03-20");

  return (
    <div className="flex flex-col gap-2">
      <p className="text-sm">你选择了范围日期：{value}</p>
      <DateRangeCalendar value={value} onChange={setValue}>
        <div className="flex gap-1">
          <CalendarPrev />
          <CalendarTitle />
          <CalendarNext />
        </div>
        <CalendarGrid />
      </DateRangeCalendar>
    </div>
  );
}
```

### 月选择

```jsx demo
import {
  DateRangeCalendar,
  CalendarPrev,
  CalendarNext,
  CalendarTitle,
  CalendarGrid,
} from "@resolid/react-ui";

export default function App() {
  return (
    <DateRangeCalendar isDateDisabled={(d) => d.getMonth() === 2} view="year" format="YYYY-MM">
      <div className="flex gap-1">
        <CalendarPrev />
        <CalendarTitle />
        <CalendarNext />
      </div>
      <CalendarGrid />
    </DateRangeCalendar>
  );
}
```

### 年选择

```jsx demo
import {
  DateRangeCalendar,
  CalendarPrev,
  CalendarNext,
  CalendarTitle,
  CalendarGrid,
} from "@resolid/react-ui";

export default function App() {
  return (

        <CalendarPrev />
        <CalendarTitle />
        <CalendarNext />
      </div>
      <CalendarGrid />
    </DateRangeCalendar>
  );
}
```

### 无效日期

```jsx demo
import {
  DateRangeCalendar,
  CalendarPrev,
  CalendarNext,
  CalendarTitle,
  CalendarGrid,
} from "@resolid/react-ui";
import { isWeekend } from "@resolid/utils/date";

export default function App() {
  return (
    <DateRangeCalendar isDateUnavailable={(d) => isWeekend(d)}>
      <div className="flex gap-1">
        <CalendarPrev />
        <CalendarTitle />
        <CalendarNext />
      </div>
      <CalendarGrid />
    </DateRangeCalendar>
  );
}
```
